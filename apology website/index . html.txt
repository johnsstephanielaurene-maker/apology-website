  <script>
    // Countdown timer
    const startDate = new Date("August 10, 2025 20:00:00").getTime();
    setInterval(function() {
      let now = new Date().getTime();
      let distance = now - startDate;

      let days = Math.floor(distance / (1000 * 60 * 60 * 24));
      let hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
      let minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
      let seconds = Math.floor((distance % (1000 * 60)) / 1000);

      document.getElementById("time").innerHTML = 
        days + "d " + hours + "h " + minutes + "m " + seconds + "s ";
    }, 1000);

    // Days missed counter
    const lastSeenDate = new Date("August 14, 2025"); // Date of the fight
    function updateMissedDays() {
      const now = new Date();
      const diffTime = now - lastSeenDate;
      const diffDays = Math.floor(diffTime / (1000 * 60 * 60 * 24));
      document.getElementById("missed-days").textContent = diffDays;
    }
    updateMissedDays();
    setInterval(updateMissedDays, 3600000); // updates every hour

    // Music control
    const music = document.getElementById("bg-music");
    function toggleMusic() {
      if (music.paused) {
        music.play();
      } else {
        music.pause();
      }
    }

    // Heart animation on button click
    const button = document.querySelector('.button');

    button.addEventListener('click', function(e) {
      const heart = document.createElement('div');
      heart.classList.add('heart');
      heart.style.left = e.clientX + 'px';
      heart.style.top = e.clientY + 'px';
      heart.textContent = '❤';
      document.body.appendChild(heart);

      setTimeout(() => {
        heart.remove();
      }, 1500);
    });
  </script>
